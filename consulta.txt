select ac.first_name, ac.last_name, f.title, count(cus.customer_id) espectadores from actor ac join film_actor fa on ac.actor_id = fa.actor_id and fa.actor_id = 3 join film f on fa.film_id = f.film_id join inventory i on f.film_id = i.film_id join rental r on i.inventory_id = r.inventory_id join customer cus on cus.customer_id = r.customer_id group by f.title order by espectadores desc, f.title;
